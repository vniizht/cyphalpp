35  QVariant getName(const E& e) const {
36  auto pos = std::find(values.begin(), values.end(), e.*field);
37  if(pos == values.end()){
38  return QVariant{QStringLiteral("Unknown value %1").arg(e.*field)};
39  }
40  auto i = pos - values.begin();
41  return names[i];
42  }
43  QVariant getDescription(const E& e) const {
44  auto pos = std::find(values.begin(), values.end(), e.*field);
45  if(pos == values.end()){
46  return QVariant{QStringLiteral("Unknown value %1").arg(e.*field)};
47  }
48  auto i = pos - values.begin();
49  return descriptions[i];
50  }
51  E getValue(int i) const {
52  E ret{};
53  if(i>=0 and static_cast<size_t>(i)< values.size()){
54  ret.*field = values.at(i);
55  }

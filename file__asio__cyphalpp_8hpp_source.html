80 class AsioUdpSocket final: public UdpSocketImpl{
81  cyphalpp::packetRecievedHandler handler_{};
82  udp::socket socket_;
83  udp::endpoint sender_endpoint_{};
84  MessageAddr base_{0};
85  uint8_t data_[UDP_MAX_MTU]{};
86  bool write_bound_{false};
87 public:
88  AsioUdpSocket(io_service& service):UdpSocketImpl(), socket_(service, udp::v4()){
89  }
90  ~AsioUdpSocket() override;
91  virtual void prepare(const MessageAddr& base, packetRecievedHandler handler) override{
92  handler_ = handler;
93  base_ = base;
94  }
95 
96  // this function MUST copy data into its own buffers
97  virtual size_t writeTo(const NetworkAddress& addr, const uint8_t*const data, size_t size)override{
98  if(not write_bound_){
99  auto base_v4 =ip::make_address_v4(base_);
100  socket_.bind(udp::endpoint(base_v4, 0));
101  socket_.set_option(ip::multicast::outbound_interface(base_v4));
102  socket_.set_option(ip::multicast::enable_loopback(false));
103  write_bound_ = true;
104  }
105  return socket_.send_to(buffer(data, size), udp::endpoint(ip::make_address_v4(addr.addr), addr.port));
106  }
107  virtual void listenUnicast(const NetworkAddress& addr)override{
108  udp::endpoint listen_endpoint(ip::make_address_v4(addr.addr), addr.port);
109  socket_.set_option(ip::multicast::enable_loopback(false));
110  socket_.set_option(udp::socket::reuse_address(true));
111  socket_.bind(listen_endpoint);
112  write_bound_ = true;
113  restartReceiving();
114  }
115  virtual void listenMulticast(const NetworkAddress& addr) override{
116  auto base_addr_v4 = ip::make_address_v4(base_);
117  auto addr_v4 = ip::make_address_v4(addr.addr);
118  socket_.set_option(udp::socket::reuse_address(true));
119  udp::endpoint listen_endpoint(addr_v4, addr.port);
120  socket_.bind(listen_endpoint);
121  socket_.set_option(ip::multicast::enable_loopback(false));
122  write_bound_ = true;
123  // Join the multicast group.
124  socket_.set_option(ip::multicast::join_group(addr_v4, base_addr_v4));
125  restartReceiving();
126  }
127 private:
128  void restartReceiving(){
129  socket_.async_receive_from(
130  boost::asio::buffer(data_, UDP_MAX_MTU), sender_endpoint_,
131  boost::bind(
132  &AsioUdpSocket::handle_receive_from, this,
133  boost::asio::placeholders::error,
134  boost::asio::placeholders::bytes_transferred));
135  }
136  void handle_receive_from(const boost::system::error_code& error, size_t bytes_recvd)
137  {
138  if(error){
139  std::clog << "Error receive" << error;
140  return;
141  }
142  if(handler_ == nullptr) return;
143 
144  cyphalpp::NetworkAddress senderAddr{
145  cyphalpp::MessageAddr(sender_endpoint_.address().to_v4().to_uint()),
146  static_cast<uint16_t>(sender_endpoint_.port())
147  };
148  auto local = socket_.local_endpoint();
149  cyphalpp::NetworkAddress recieverAddr{
150  cyphalpp::MessageAddr(local.address().to_v4().to_uint()),
151  static_cast<uint16_t>(local.port())
152  };
153  handler_(senderAddr, recieverAddr, data_, bytes_recvd);
154  restartReceiving();
155  }
156 };

61 struct MessageAddr{
62 public:
63  constexpr MessageAddr(uint8_t subnet_id, uint16_t subject_id)
64  :value_(0b111011110 << 23){
65  value_ |= ( (static_cast<uint32_t>(subnet_id) & 0x7FU ) << 16 );
66  value_ |= (static_cast<uint32_t>(subject_id) & 0x1FFFU );
67  }
68  constexpr MessageAddr(uint16_t prefix_9bits, uint8_t subnet_id, uint16_t subject_id)
69  :value_(0){
70  value_ |= (prefix_9bits & 0x1FFU) << 23;
71  value_ |= ( (static_cast<uint32_t>(subnet_id) & 0x7FU ) << 16 );
72  value_ |= (static_cast<uint32_t>(subject_id) & 0x1FFFU );
73  }
75  MessageAddr(uint32_t addr) : value_(addr) { }
76  constexpr uint32_t prefix_9bit() const{ return (value_ >> 23) & 0x1FF; }
77  constexpr uint8_t subnet_id() const { return static_cast<uint8_t>((value_ >> 16) & 0x7FU); }
79  constexpr uint16_t subject_or_node_id() const { return static_cast<uint16_t>(value_ & 0x1FFFU); }
81  constexpr bool isValidMessage() const{ return prefix_9bit() == 0b111011110; }
83  operator uint32_t()const { return value_; }
84 private:
85  uint32_t value_;
86 };

137 inline tl::expected<DataSpecifier, Error> DataSpecifier::from_packet(
138  const MessageAddr& currentAddr, const NetworkAddress& source, const NetworkAddress& dest){
139  if(not (
140  source.addr.subnet_id() == currentAddr.subnet_id()
141  and dest.addr.subnet_id() == currentAddr.subnet_id()
142  )){
143  return -errors::DataSpecifierConversion::WrongSubnet;
144  }
145  if(dest.addr.isValidMessage()){
146  if(dest.port == MULTICAST_PORT){
147  return DataSpecifier{
148  DataSpecifier::Message,
149  source.addr.subject_or_node_id(),
150  dest.addr.subject_or_node_id(),
151  };
152  }
153  return -errors::DataSpecifierConversion::WrongPort;
154  }
155  if (dest.port >= UNICAST_BASE_PORT){
156  return DataSpecifier{
157  (dest.port % 2 == 0)?(DataSpecifier::ServiceRequest):(DataSpecifier::ServiceResponce),
158  source.addr.subject_or_node_id(),
159  static_cast<uint16_t>((dest.port - UNICAST_BASE_PORT) / 2U),
160  };
161  }
162  return -errors::DataSpecifierConversion::PortTooSmall;

529  std::unique_ptr<UdpSocketImpl> createSocket(Subscription& sub){
530  auto udp = socketFactory();
531  udp->prepare(addr_, [this, &sub](
532  const NetworkAddress& source, const NetworkAddress& dest,
533  const uint8_t*const data, size_t size
534  ) -> tl::expected<void, Error> {
535  if(size < cyphal_udp::Header_1_0::EXTENT_BYTES){
536  return -errors::ParsePacket::MessageTooSmall;
537  }
538  if(source.addr == addr_ and not loopback_enabled){
539  return {};
540  }
541  TransferMetadata tm;
542  {
543  auto maybe_ds = DataSpecifier::from_packet(addr_, source, dest);
544  if(not maybe_ds){
545  return tl::unexpected<Error>(maybe_ds.error());
546  }
547  tm.data = maybe_ds.value();
548  }
549  if(sub.ds.role != tm.data.role){
550  return -errors::ParsePacket::RolesMismatch;
551  }
552  if(sub.ds.subject_id != tm.data.subject_id){
553  return -errors::ParsePacket::SubjectIdMismatch;
554  }
555  cyphal_udp::Header_1_0 header;
556  auto headerResult = header.deserialize({data, size});
557  if(not headerResult){
558  return tl::unexpected<Error>(headerResult.error());
559  }
560  if(header.version != 0 ){
561  return -errors::ParsePacket::HeaderWrongProtocolVersion;
562  }
563  tm.transfer_id = header.transfer_id;
564  tm.priority = header.priority;
565  if(header.frame_index_eot != cyphal_udp::Header_1_0::EndOfTransmission){
566  return -errors::ParsePacket::MultiframeTransfersNotSupported;
567  }
568  return sub.handler(&sub, tm, data + cyphal_udp::Header_1_0::EXTENT_BYTES, size - cyphal_udp::Header_1_0::EXTENT_BYTES);
569  });
570  return udp;
571  }

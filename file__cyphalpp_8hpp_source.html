61 struct MessageAddr{
62 public:
63  constexpr MessageAddr(uint8_t subnet_id, uint16_t subject_id)
64  :value_(0b111011110 << 23){
65  value_ |= ( (static_cast<uint32_t>(subnet_id) & 0x7FU ) << 16 );
66  value_ |= (static_cast<uint32_t>(subject_id) & 0x1FFFU );
67  }
68  constexpr MessageAddr(uint16_t prefix_9bits, uint8_t subnet_id, uint16_t subject_id)
69  :value_(0){
70  value_ |= (prefix_9bits & 0x1FFU) << 23;
71  value_ |= ( (static_cast<uint32_t>(subnet_id) & 0x7FU ) << 16 );
72  value_ |= (static_cast<uint32_t>(subject_id) & 0x1FFFU );
73  }
75  MessageAddr(uint32_t addr) : value_(addr) { }
77  constexpr uint32_t prefix_9bit() const{ return (value_ >> 23) & 0x1FF; }
79  constexpr uint8_t subnet_id() const { return static_cast<uint8_t>((value_ >> 16) & 0x7FU); }
81  constexpr uint16_t subject_or_node_id() const { return static_cast<uint16_t>(value_ & 0x1FFFU); }
83  constexpr bool isValidMessage() const{ return prefix_9bit() == 0b111011110; }
85  operator uint32_t()const { return value_; }
86 private:
87  uint32_t value_;
88 };

142 inline tl::expected<DataSpecifier, Error> DataSpecifier::from_packet(
143  const MessageAddr& currentAddr, const NetworkAddress& source, const NetworkAddress& dest){
144  if(not (
145  source.addr.subnet_id() == currentAddr.subnet_id()
146  and dest.addr.subnet_id() == currentAddr.subnet_id()
147  )){
148  return -errors::DataSpecifierConversion::WrongSubnet;
149  }
150  if(dest.addr.isValidMessage()){
151  if(dest.port == MULTICAST_PORT){
152  return DataSpecifier{
153  DataSpecifier::Message,
154  source.addr.subject_or_node_id(),
155  dest.addr.subject_or_node_id(),
156  };
157  }
158  return -errors::DataSpecifierConversion::WrongPort;
159  }
160  if (dest.port >= UNICAST_BASE_PORT){
161  return DataSpecifier{
162  (dest.port % 2 == 0)?(DataSpecifier::ServiceRequest):(DataSpecifier::ServiceResponce),
163  source.addr.subject_or_node_id(),
164  static_cast<uint16_t>((dest.port - UNICAST_BASE_PORT) / 2U),
165  };
166  }
167  return -errors::DataSpecifierConversion::PortTooSmall;

557  std::unique_ptr<UdpSocketImpl> createSocket(Subscription& sub){
558  auto udp = socketFactory();
559  udp->prepare(addr_, [this, &sub](
560  const NetworkAddress& source, const NetworkAddress& dest,
561  const uint8_t*const data, size_t size
562  ) -> tl::expected<void, Error> {
563  if(size < cyphal_udp::Header_1_0::EXTENT_BYTES){
564  return -errors::ParsePacket::MessageTooSmall;
565  }
566  if(source.addr == addr_ and not loopback_enabled){
567  return {};
568  }
569  TransferMetadata tm;
570  {
571  auto maybe_ds = DataSpecifier::from_packet(addr_, source, dest);
572  if(not maybe_ds){
573  return tl::unexpected<Error>(maybe_ds.error());
574  }
575  tm.data = maybe_ds.value();
576  }
577  if(sub.ds.role != tm.data.role){
578  return -errors::ParsePacket::RolesMismatch;
579  }
580  if(sub.ds.subject_id != tm.data.subject_id){
581  return -errors::ParsePacket::SubjectIdMismatch;
582  }
583  cyphal_udp::Header_1_0 header;
584  auto headerResult = header.deserialize({data, size});
585  if(not headerResult){
586  return tl::unexpected<Error>(headerResult.error());
587  }
588  if(header.version != 0 ){
589  return -errors::ParsePacket::HeaderWrongProtocolVersion;
590  }
591  tm.transfer_id = header.transfer_id;
592  tm.priority = header.priority;
593  if(header.frame_index_eot != cyphal_udp::Header_1_0::EndOfTransmission){
594  return -errors::ParsePacket::MultiframeTransfersNotSupported;
595  }
596  return sub.handler(&sub, tm, data + cyphal_udp::Header_1_0::EXTENT_BYTES, size - cyphal_udp::Header_1_0::EXTENT_BYTES);
597  });
598  return udp;
599  }
